#!/bin/bash
#星座的名字,抓网页的时候要用的
Signs="aries taurus gemini cancer leo virgo libra scorpio sagittarius capricorn aquarius pisces"
#日期,每篇文章标题要用的
head=`date --date='1 days' +%Y-%m-%d`
#要贴的BBS站点,都是水木代码的
Sites='bbs.buaa.edu.cn'
#用户名密码
user=horoscope
pass=*****
#生成每个星座的星运文件
i=1;
for sign in $Signs
do
wget http://mindcity.sina.com.tw/west/MC-12stars/${sign}2.html -O $i.htm
sed -n '//,//p' $i.htm | sed -e 's/
/\n/g' -e 's/<[^>]*>//g' -e 's/\t//'| sed -n '4,$p' | sed -e '2,4d' -e '15d'|iconv -f BIG5 -t GB2312 > $i.ast
rm $i.htm
let i=$i+1
done
#out 用户名 密码 文件
out(){
sleep 1
echo "$1"
sleep 1
echo "$2"
echo
sleep 1
echo -n "qqqqqqqqqq"
sleep 1
echo "S"
echo "horoscope"
echo
sleep 3
j=1
while [ $j -lt 13 ]
do
echo -n $'\cp'
((echo -n $head) && (head -1 $j.ast))
echo
echo $head
echo
cat $j.ast
echo
echo -n $'\cx'
echo
let j=$j+1
sleep 3
done
sleep 3
echo "qqqqg"
echo
echo
echo
}
for site in $Sites
do
out $user $pass | telnet -8 $site
done
